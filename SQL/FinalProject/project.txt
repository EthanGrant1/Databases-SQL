 16  --
 17  -- properStoreSize: Store sizes must be small, medium, or large

SQL> CREATE TABLE OrderLines (
  2  lineID INTEGER NOT NULL,
  3  orderNum INTEGER NOT NULL,
  4  itemOrdered INTEGER NOT NULL,
  5  qty INTEGER NOT NULL,
  6  storeNum INTEGER NOT NULL,
  7  shipmentType CHAR(25) NOT NULL,
  8  price CHAR(15) NOT NULL,
  9  daysToShip INTEGER NOT NULL,
 10  --
 11  -- Integrity Constraints
 12  --
 13  -- orderLineCK: OrderLines have a composite key of both the lineID and the orderNum
 14  CONSTRAINT orderLineCK PRIMARY KEY (lineID, orderNum),
 15  --
 16  -- orderLineIDLen: Order Line IDs are 6 digits long
 17  CONSTRAINT orderLineIDLen CHECK ((lineID >= 100000) AND (lineID <= 999999)),
 18  --
 19  -- orderLineOrderFK: orderNum is a foreign key that references Orders(orderNum)
 20  CONSTRAINT orderLineOrderFK FOREIGN KEY (orderNum) REFERENCES Orders (orderNum)
 21  ON DELETE CASCADE,
 22  --
 23  -- orderLineItemNameFK: itemOrdered is a foreign key that references Items(serialNum)
 24  CONSTRAINT orderLineItemNameFK FOREIGN KEY (itemOrdered) REFERENCES Items(serialNum)
 25  ON DELETE CASCADE,
 26  --
 27  -- orderLineQty: qty is greater than 0
 28  CONSTRAINT orderLineQty CHECK (qty > 0),
 29  --
 30  -- orderLineStoreFK: storeNum is a foreign key that references Stores(storeNum)
 31  CONSTRAINT orderLineStoreFK FOREIGN KEY (storeNum) REFERENCES Stores(storeNum)
 32  ON DELETE CASCADE,
 33  --
 34  -- orderLineShipmentType: shipmentType must be 'regular', 'express', or 'same-day'
 35  CONSTRAINT orderLineShipmentType CHECK (shipmentType IN ('regular', 'express', 'same-day')),
 36  --
 37  -- orderLinePriceSyntax: OrderLine costs are of the form $x.xx (ex: $5.99, $375.55, etc)
 38  CONSTRAINT orderLinePriceSyntax CHECK (price LIKE '$%.%'),
 39  --
 40  -- orderLineShipmentTime: Different shipment types have differing amounts of shipping time (regular: 5+ days, express: 3 days, same-day: 1 day)
 41  --
 42  CONSTRAINT orderLineShipmentTime
 43  CHECK ((shipmentType = 'regular' AND daysToShip >= 5)   OR
 44  (shipmentType = 'express' AND daysToShip = 3)   OR
 45  (shipmentType = 'same-day' AND daysToShip = 1))
 46  );
